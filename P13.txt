// File: apack/A.java
package apack;

public class A {
    public int pubVar;
    protected int protVar;
    private int privVar;

    public A(int pubVar, int protVar, int privVar) {
        this.pubVar = pubVar;
        this.protVar = protVar;
        this.privVar = privVar;
    }
}



// File: bpack/B.java
package bpack;

import apack.A;

public class B extends A {
    
    public B(int pubVar, int protVar, int privVar) {
        super(pubVar, protVar, privVar);
    }

    public void display() {
        System.out.println("Inside Class B (Subclass in bpack):");
        System.out.println("Public Variable: " + pubVar); 
        System.out.println("Protected Variable: " + protVar); 
        // privVar is private in A, so it cannot be accessed here.
    }
}




// File: cpack/C.java
package cpack;

import apack.A;

public class C {
    
    public void display() {
        A objA = new A(10, 20, 30);
        System.out.println("Inside Class C (Non-subclass in cpack):");
        System.out.println("Public Variable: " + objA.pubVar);
        // protVar is protected in A, so it cannot be accessed here.
        // privVar is private in A, so it cannot be accessed here.
    }
}




// File: dpack/ProtectedDemo.java
package dpack;

import bpack.B;
import cpack.C;

public class ProtectedDemo {
    public static void main(String[] args) {
        B objB = new B(100, 200, 300);
        objB.display();

        System.out.println("--------------------");

        C objC = new C();
        objC.display();
    }
}
