import java.util.*;

class University {

    static int totalStudents;
    static String universityName;

    // Static Block (runs once)
    static {
        universityName = "ABC University";
        totalStudents = 0;
        System.out.println("Static Block Executed");
        System.out.println("University Name: " + universityName);
        System.out.println();
    }

    // Instance Block (runs every time object is created)
    {
        System.out.println("Instance Block Executed - Student Object Created");
    }

    // Constructor
    University() {
        totalStudents++;
        System.out.println("Constructor Executed");
        System.out.println();
    }

    // Static Method
    static void getTotalStudents() {
        System.out.println("Total Students Created: " + totalStudents);
    }
}

public class P8 {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.println("Main Method Started\n");

        System.out.print("Enter number of students to create: ");
        int n = sc.nextInt();

        for (int i = 0; i < n; i++) {
            new University();   // object creation
        }

        University.getTotalStudents();
    }
}
