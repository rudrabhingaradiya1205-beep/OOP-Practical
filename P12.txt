import java.util.Scanner;

interface Order {
    void placeOrder(String item, int qty);
    void cancelOrder(int orderId);
    void generateBill();
}

abstract class PartialOrder implements Order {
    String itemName;
    int quantity;
    int orderId;

    public void placeOrder(String item, int qty) {
        this.itemName = item;
        this.quantity = qty;
        this.orderId = 101; 
        System.out.println("Order placed! Your Order ID is: " + orderId);
    }
}

class FinalOrder extends PartialOrder {
    int pricePerItem = 150; 

    public void cancelOrder(int id) {
        if (this.orderId == id) {
            System.out.println("Order ID " + id + " has been cancelled successfully.");
            this.quantity = 0; 
        } else {
            System.out.println("Invalid Order ID.");
        }
    }

    public void generateBill() {
        if (quantity > 0) {
            int totalAmount = quantity * pricePerItem;
            System.out.println("--- Bill ---");
            System.out.println("Item: " + itemName);
            System.out.println("Quantity: " + quantity);
            System.out.println("Total Amount: Rs." + totalAmount);
        } else {
            System.out.println("No active order found to generate a bill.");
        }
    }
}

public class P12 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        FinalOrder order = new FinalOrder();

        System.out.print("Enter item to order: ");
        String item = scanner.nextLine();
        
        System.out.print("Enter quantity: ");
        int qty = scanner.nextInt();

        order.placeOrder(item, qty);

        System.out.println("\nWhat would you like to do next?");
        System.out.println("1. Generate Bill");
        System.out.println("2. Cancel Order");
        System.out.print("Enter choice (1 or 2): ");
        int choice = scanner.nextInt();

        if (choice == 1) {
            order.generateBill();
        } else if (choice == 2) {
            System.out.print("Enter your Order ID to confirm cancellation: ");
            int id = scanner.nextInt();
            order.cancelOrder(id);
        } else {
            System.out.println("Invalid choice.");
        }

        scanner.close();
    }
}
