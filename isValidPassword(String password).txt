import java.util.*;

public class P9 {

    public static boolean passwordIsVlid(String password) {

        if (password.length() < 8) {
            System.out.println("AT LEAST 8 CHARACTERS");
            return false;
        }

        int digitCount = 0;

        for (int i = 0; i < password.length(); i++) {
            char ch = password.charAt(i);

            if (!Character.isLetterOrDigit(ch)){
                return false;
            }

            if (Character.isDigit(ch)){
                digitCount++;
            }
        }

        return digitCount >= 2;
    }

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.println("S.S.A.S.I.T.");
        System.out.println("COMPUTER ENGINEERING DEPARTMENT\n");

        System.out.println("RULES - PASSWORD\n");
        System.out.println("1. ATLEAST 8 CHARACTERS");
        System.out.println("2. LETTERS AND DIDGITS ONLY");
        System.out.println("3. ATLEAST TWO DIGITS\n");

        System.out.print("ENTER YOUR PASSWORD :- ");
        String password = sc.nextLine();

        if (passwordIsVlid(password)){
            System.out.println("VALID PASSWORD...");
        } else{
            System.out.println("INVALID PASSWORD !!!");
        }
    }
}
