import java.util.Scanner;

class Payment {
    public void processPayment(int amount) {
        System.out.println("Payment of Rs." + amount + " being processed...");
    }
}

class CreditCardPayment extends Payment {
    @Override
    public void processPayment(int amount) {
        System.out.println("Payment of Rs." + amount + " done using Credit Card");
    }
}

class UPIPayment extends Payment {
    @Override
    public void processPayment(int amount) {
        System.out.println("Payment of Rs." + amount + " done using UPI");
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter payment amount: ");
        int amount = scanner.nextInt();

        System.out.println("Select Payment Method:");
        System.out.println("1. Credit Card");
        System.out.println("2. UPI");
        System.out.print("Enter your choice: ");
        int choice = scanner.nextInt();

        Payment paymentMethod;

        if (choice == 1) {
            paymentMethod = new CreditCardPayment();
        } else if (choice == 2) {
            paymentMethod = new UPIPayment();
        } else {
            System.out.println("Invalid choice. Transaction cancelled.");
            scanner.close();
            return;
        }

        paymentMethod.processPayment(amount);

        scanner.close();
    }
}
