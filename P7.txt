import java.util.*;

class LoanCalculator {

    Scanner sc = new Scanner(System.in);

    // Home Loan
    void calculateEMI(int principal, int time, float rate) {
        float interest = (principal * rate * time) / 100;
        float totalAmount = principal + interest;
        float emi = totalAmount / (time * 12);
        System.out.println("Home Loan EMI: ₹" + emi);
    }

    // Vehicle Loan
    void calculateEMI(double principal, int time, double rate) {
        double interest = (principal * rate * time) / 100;
        double totalAmount = principal + interest;
        double emi = totalAmount / (time * 12);
        System.out.println("Vehicle Loan EMI: ₹" + emi);
    }

    // Personal Loan (Fixed 10%)
    void calculateEMI(int principal, int time) {
        float rate = 10;  
        float interest = (principal * rate * time) / 100;
        float totalAmount = principal + interest;
        float emi = totalAmount / (time * 12);
        System.out.println("Personal Loan EMI (10% fixed): ₹" + emi);
    }

    void takeInput() {
        System.out.println("Select Loan Type:");
        System.out.println("1. Home Loan");
        System.out.println("2. Vehicle Loan");
        System.out.println("3. Personal Loan");
        
        System.out.print("Enter Your Choice :- ");
        int choice = sc.nextInt();

        if (choice == 1) {
            System.out.print("Enter Principal: ");
            int p = sc.nextInt();
            System.out.print("Enter Time (years): ");
            int t = sc.nextInt();
            System.out.print("Enter Rate: ");
            float r = sc.nextFloat();
            calculateEMI(p, t, r);

        } else if (choice == 2) {
            System.out.print("Enter Principal: ");
            double p = sc.nextDouble();
            System.out.print("Enter Time (years): ");
            int t = sc.nextInt();
            System.out.print("Enter Rate: ");
            double r = sc.nextDouble();
            calculateEMI(p, t, r);

        } else if (choice == 3) {
            System.out.print("Enter Principal: ");
            int p = sc.nextInt();
            System.out.print("Enter Time (years): ");
            int t = sc.nextInt();
            calculateEMI(p, t);

        } else {
            System.out.println("Invalid Choice");
        }
    }
}

public class P7 {
    public static void main(String[] args) {

        LoanCalculator loan = new LoanCalculator();
        loan.takeInput();
    }
}
