import java.io.*;
import java.util.Date;

public class Main {
    public static void main(String[] args) {
        if (args.length < 2) {
            System.out.println("Usage:");
            System.out.println("  java FileManager copy <source> <destination>");
            System.out.println("  java FileManager delete <filename>");
            System.out.println("  java FileManager rename <oldname> <newname>");
            return;
        }

        String command = args[0].toLowerCase();
        File file1 = new File(args[1]);

        try {
            switch (command) {
                case "copy":
                    if (args.length < 3) throw new IOException("Destination file missing!");
                    File destFile = new File(args[2]);
                    copyFile(file1, destFile);
                    System.out.println("SUCCESS: File copied.");
                    displayProperties(destFile);
                    break;

                case "delete":
                    if (!file1.exists()) throw new FileNotFoundException("File not found: " + file1.getName());
                    if (file1.delete()) {
                        System.out.println("SUCCESS: File '" + file1.getName() + "' deleted.");
                    } else {
                        System.out.println("FAILED: Could not delete the file.");
                    }
                    break;

                case "rename":
                    if (args.length < 3) throw new IOException("New name missing!");
                    File renamedFile = new File(args[2]);
                    if (file1.renameTo(renamedFile)) {
                        System.out.println("SUCCESS: File renamed.");
                        displayProperties(renamedFile);
                    } else {
                        throw new IOException("Rename failed (check if source exists).");
                    }
                    break;

                default:
                    System.out.println("Error: Unknown command '" + command + "'");
            }
        } catch (FileNotFoundException e) {
            System.out.println("FILE NOT FOUND ERROR: " + e.getMessage());
        } catch (IOException e) {
            System.out.println("IO ERROR: " + e.getMessage());
        }
    }

    private static void copyFile(File src, File dest) throws IOException {
        if (!src.exists()) throw new FileNotFoundException("Source file does not exist.");

        try (FileInputStream in = new FileInputStream(src);
             FileOutputStream out = new FileOutputStream(dest)) {
            byte[] buffer = new byte[1024];
            int length;
            while ((length = in.read(buffer)) > 0) {
                out.write(buffer, 0, length);
            }
        }
    }

    private static void displayProperties(File f) {
        System.out.println("\n--- FILE PROPERTIES ---");
        System.out.println("1. File Name:      " + f.getName());
        System.out.println("2. Absolute Path:  " + f.getAbsolutePath());
        System.out.println("3. File Size:      " + f.length() + " bytes");
        System.out.println("4. Readable:       " + f.canRead());
        System.out.println("   Writable:       " + f.canWrite());
        System.out.println("5. Last Modified:  " + new Date(f.lastModified()));
        System.out.println("------------------------\n");
    }
}
